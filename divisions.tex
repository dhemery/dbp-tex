% How many lines down the page to place \sink titles.
\newcount\sinklines

\def\frontmatter{\nextrectopage\normalbaselines\global\pageno=-1 \useemptypageframe}
\def\mainmatter{\nextrectopage\normalbaselines\global\pageno=1 \usenormalpageframe}
\def\backmatter{\frontmatter}
\def\endbook{\nextrectopage\end}

\def\part#1{
  \nextpartpage
  \sink{\raggedspine\parttitle{#1}}
  \nextrectopage
}
\def\nextpartpage{\nextrectopage}

\def\chapter#1{
  \nextchapterpage
  \usedisplayframethispage
  \sink{\raggedspine\chaptertitle{#1}}
}
\def\nextchapterpage{\nextpage}
\def\rectochapters{\gdef\nextchapterpage{\nextrectopage}}

% \bookinfo is a title division in frontmatter or backmatter.
\def\bookinfo#1{
  \nextpage
  \useemptypageframe
  {\centered\chaptertitle{#1}}
}

% Boxes the given content and sinks its baseline \sinklines
% text lines down the page.
%
% \sink works only when it is the first content added to a new page.
\long\def\sink#1{{
  \topskip=\sinklines\leading
    \advance\topskip by -\maxdepth
  \vbox{#1}
}}

\def\booktitle#1{{\style{booktitle}\tightbaselines\uppercase\expandafter{#1}\par}}
\def\authorname#1{{\style{authorname}\tightbaselines\uppercase\expandafter{#1}\par}}
\def\parttitle#1{{\style{parttitle}\sc\tightbaselines#1\par}}
\def\chaptertitle#1{{\style{chaptertitle}\sc\tightbaselines#1\par}}

\sinklines=11
