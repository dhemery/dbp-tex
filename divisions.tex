\newtoks\sinkratio
\newdimen\sinkheight
\countdef\partno=2
\countdef\chapterno=3

\def\frontmatter{
  \global\pageno=1
  \global\pagecount=1
  \normalpages
  \emptypageframes
  \normalbaselines
}

\def\mainmatter{
  \nextrectopage
  \global\pageno=1
  \normalpages
  \normalpageframes
  \normalbaselines
}

\def\backmatter{
  \nextrectopage
  \global\pageno=\pagecount
  \normalpages
  \emptypageframes
  \normalbaselines
}

\def\endbook{\nextrectopage\end}

\def\part#1{
  \global\advance\partno by 1
  \partpage
  \partpageframe
  \partheading{#1}
  \afterpartheading
}
\def\partpage{\nextrectopage}
\def\partpageframe{\emptyframethispage}
\def\partheading#1{\sink{\heading{\parttitlefont}{#1}}}
\def\afterpartheading{\nextrectopage}

\def\chapter#1{
  \global\advance\chapterno by 1
  \chapterpage
  \chapterpageframe
  \chapterheading{#1}
  \afterchapterheading
}
\def\chapterpage{\nextpage}
\def\chapterpageframe{\displayframethispage}
\def\chapterheading#1{\sink{\heading{\chaptertitlefont}{#1}}}
\def\afterchapterheading{\skipline}

% \bookinfo is division in frontmatter or backmatter.
\def\bookinfo#1{
  \bookinfopage
  \bookinfopageframe
  \bookinfoheading{#1}
  \afterbookinfoheading
}
\def\bookinfopage{\nextpage}
\def\bookinfopageframe{}
\def\bookinfotitlefont{\chaptertitlefont}
\def\bookinfoheading#1{\baselinebox{\centeredheading{\bookinfotitlefont}{#1}}}
\def\afterbookinfoheading{\skipline}

\def\sink#1{\vbox to \sinkheight{\vfil#1}}

\def\setsink#1{
  \global\sinkratio={#1}
  \sinkheight=\the\textlines\leading % text extent
  \sinkheight=\the\sinkratio\sinkheight % unaligned sink position
  \advance\sinkheight by .5\leading % half a line for rounding
  \divide\sinkheight by \leading % sink line number
  \multiply\sinkheight by \leading % bottom of depth of sink line
  \advance\sinkheight by -\maxdepth % baseline of sink line
  \showsink
}

\def\showsink{
  \showparam\sinkratio
  \showparam\sinkheight
}
