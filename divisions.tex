% How many lines down the page to place \sink titles.
\newcount\sinklines
\countdef\partno=2
\countdef\chapterno=3

\def\frontmatter{
  \global\pageno=1
  \global\pagecount=1
  \normalpages
  \emptypageframes
  \normalbaselines
}

\def\mainmatter{
  \nextrectopage
  \global\pageno=1
  \normalpages
  \normalpageframes
  \normalbaselines
}

\def\backmatter{
  \nextrectopage
  \global\pageno=\pagecount
  \normalpages
  \emptypageframes
  \normalbaselines
}

\def\endbook{\nextrectopage\end}

\def\part#1{
  \global\advance\partno by 1
  \partpage
  \partheading{#1}
  \afterpartheading
}
\def\partpage{\nextrectopage \emptyframethispage}
\def\partheading#1{\sink{\heading{\parttitletype}{#1}}}
\def\afterpartheading{\nextrectopage}

\def\chapter#1{
  \global\advance\chapterno by 1
  \chapterpage
  \chapterheading{#1}
  \afterchapterheading
}
\def\chapterpage{\nextpage \displayframethispage}
\def\chapterheading#1{\sink{\heading{\chaptertitletype}{#1}}}
\def\afterchapterheading{\skipline}
\def\rectochapters{\gdef\chapterpage{\nextrectopage}}

% \bookinfo is division in frontmatter or backmatter.
\def\bookinfo#1{
  \bookinfopage
  \bookinfoheading{#1}
  \afterbookinfoheading
}
\def\bookinfopage{\nextpage}
\def\bookinfotitletype{\chaptertitletype}
\def\bookinfoheading#1{\centeredheading{\bookinfotitletype}{#1}}
\def\afterbookinfoheading{\skipline}

% Boxes the given content and sinks its baseline \sinklines
% text lines down the page.
%
% \sink works only when it is the first content added to a new page.
\def\sink#1{
  \dimen0=\sinklines\leading
    \advance\dimen0 by -\maxdepth
  \vbox to \dimen0{\vfil#1}
}

\sinklines=11
