\newdimen\aheadsize
\newdimen\bheadsize
\newdimen\cheadsize
\newdimen\bodytextsize
\newdimen\fineprintsize
\newdimen\runningheadsize

\def\setheadingsizes#1#2#3{
  \aheadsize=#1
  \bheadsize=#2
  \cheadsize=#3
  \setlogosize{\cheadsize}
}

\def\definestyles{%
  \newstyle{booktitle}{\booktitlefamily}{\aheadopticalsize}{tlf}{\the\aheadsize}%
  \newstyle{authorname}{\authornamefamily}{\bheadopticalsize}{tlf}{\the\bheadsize}%
  \newstyle{parttitle}{\parttitlefamily}{\aheadopticalsize}{osf}{\the\aheadsize}%
  \newstyle{chaptertitle}{\chaptertitlefamily}{\cheadopticalsize}{osf}{\the\cheadsize}%
  \newstyle{bodytext}{\bodytextfamily}{\bodytextopticalsize}{osf}{\the\bodytextsize}%
  \newstyle{fineprint}{\fineprintfamily}{\fineprintopticalsize}{tlf}{\the\fineprintsize}%
  \newstyle{runninghead}{\runningheadfamily}{\runningheadopticalsize}{osf}{\the\runningheadsize}%
}
