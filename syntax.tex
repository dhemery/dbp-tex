% Cat codes for fundamental TeX features
\catcode`\{=1 % { begin group
\catcode`\}=2 % } end group
\catcode`\$=3 % $ shift into and out of math mode
\catcode`\&=4 % & alignment tab
\catcode`\#=6 % # macro parameter prefix
\catcode`\^=7 % ^ superscript operator
\catcode`\_=8 % _ subscript operator
\catcode`\^^I=10 % Treat tab as space

% Active characters
\catcode`\~=13 \def~{\nbsp} % Treat tilde (aka "tie") as nonbreaking space
\catcode`\^^L=13 \outer\def\^^L{\par}   % Convert form feed to \par

% TODO: What other basic syntax macros should I borrow from plain TeX?
\def\'#1{{\accent"13 #1}}
\chardef\$=`\$

% \globaldefs = 0: assignments obey \global prefix
% \globaldefs < 0: all assignments are local (\global prefix ignored)
% \globaldefs > 0: all assignments are global (\global prefix implied)
\globaldefs=0
