% Garamond Premier Pro Optical Sizes
% - Caption 8 (5.9-8.9)
% - Regular 11 (8.9-14.9)
% - Subhead 18 (14.9-22.9)
% - Display 36 (22.9-72)

\def\garamond{
  \newtypeface{garamond}{GaramondPremrPro}{}{It}{Bd}{}

  \newscheme{ahead}{garamond}{Disp}{\aheadsize}
  \newscheme{bhead}{garamond}{Disp}{\bheadsize}
  \newscheme{chead}{garamond}{Subh}{\cheadsize}
  \newscheme{text}{garamond}{}{\textsize}
  \newscheme{smalltext}{garamond}{}{\smalltextsize}
  \newscheme{caption}{garamond}{Capt}{\captionsize}

  \newstyle{booktitlestyle}{ahead}{rg}{rm}{sc}{os}
  \newstyle{authornamestyle}{bhead}{rg}{rm}{sc}{os}
  \newstyle{parttitlestyle}{ahead}{rg}{rm}{sc}{os}
  \newstyle{chaptertitlestyle}{chead}{rg}{rm}{sc}{os}
  \newstyle{bodystyle}{text}{rg}{rm}{nv}{os}
  \newstyle{fineprintstyle}{smalltext}{rg}{rm}{nv}{tl}
  \newstyle{runningheadstyle}{caption}{rg}{rm}{sc}{os}
}
