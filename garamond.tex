% Garamond Premier Pro Optical Sizes
% - Caption 8 (5.9-8.9)
% - Regular 11 (8.9-14.9)
% - Subhead 18 (14.9-22.9)
% - Display 36 (22.9-72)

\def\garamond{
  \newtypeface{garamond}{GaramondPremrPro}{}{It}{Bd}{}

  \newtype{booktitle}{garamond}{Disp}{\the\aheadsize}{rg}{rm}{sc}{po}
  \newtype{authorname}{garamond}{Disp}{\the\bheadsize}{rg}{rm}{sc}{po}
  \newtype{parttitle}{garamond}{Disp}{\the\aheadsize}{rg}{rm}{sc}{po}
  \newtype{chaptertitle}{garamond}{Subh}{\the\cheadsize}{rg}{rm}{sc}{po}
  \newtype{body}{garamond}{}{\the\textsize}{rg}{rm}{nv}{po}
  \newtype{small}{garamond}{}{\the\smalltextsize}{rg}{rm}{nv}{po}
  \newtype{pageframe}{garamond}{Capt}{\the\captionsize}{rg}{rm}{sc}{po}
}

% expand body and small fonts, but not others
\def\onbodytypeload#1{\expandtextfont#1}
\def\onsmalltypeload#1{\expandtextfont#1}
