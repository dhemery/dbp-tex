% Garamond Premier Pro Optical Sizes
% - Caption 8 (5.9-8.9)
% - Regular 11 (8.9-14.9)
% - Subhead 18 (14.9-22.9)
% - Display 36 (22.9-72)

\def\garamond{
  \newtypeface{garamond}{GaramondPremrPro}{}{It}{Bd}{}

  \newscheme{ahead}{garamond}{Disp}{\aheadsize}
  \newscheme{bhead}{garamond}{Disp}{\bheadsize}
  \newscheme{chead}{garamond}{Subh}{\cheadsize}
  \newscheme{body}{garamond}{}{\textsize}
  \newscheme{small}{garamond}{}{\smalltextsize}
  \newscheme{caption}{garamond}{Capt}{\captionsize}

  \newtype{booktitle}{ahead}{rg}{rm}{sc}{os}
  \newtype{authorname}{bhead}{rg}{rm}{sc}{os}
  \newtype{parttitle}{ahead}{rg}{rm}{sc}{os}
  \newtype{chaptertitle}{chead}{rg}{rm}{sc}{os}
  \newtype{body}{body}{rg}{rm}{nv}{os}
  \newtype{small}{small}{rg}{rm}{nv}{os}
  \newtype{runninghead}{caption}{rg}{rm}{sc}{os}
}

% expand body and small fonts, but not others
\def\onbodyschemeload#1{\expandtextfont#1}
\def\onsmallschemeload#1{\expandtextfont#1}
