\newtoks\previewtitle
\newtoks\previewauthor
\newtoks\previewcontent

\def\previewtitleblock#1#2{{
  \centeredheading{\booktitlestyle}{#1}
  \skipline
  \centerline{\bodytext by}
  \skipline
  \centeredheading{\authornamestyle}{#2}
}}

\def\announce#1#2{
  \nextrectopage
  \line{}
  \vfil
  \centerline{\sc \the\publisher}
  \centerline{proudly presents}
  \vfil
  \previewtitleblock{#1}{#2}
  \vfil
  \centerline{Turn the page for a preview}
}

\def\available#1#2{
  \nextrectopage
  \line{}
  \vfil
  \previewtitleblock{#1}{#2}
  \vfil
  \centerline{Available now at your favorite retailers}
  \vfil
  \centerline{\linelogo}
}

\def\preview{
  \announce{\the\previewtitle}{\the\previewauthor}

  \nextrectopage
  \chapterheading{\the\previewtitle}
  \afterchapterheading

  \input \the\previewcontent

  \available{\the\previewtitle}{\the\previewauthor}
}

\previewtitle={\tbd\previewtitle}
\previewauthor={\tbd\previewauthor}
\previewcontent={\tbd\previewcontent}
