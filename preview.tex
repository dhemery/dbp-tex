\newtoks\previewtitle
\newtoks\previewauthor

\def\previewtitleline#1#2{{
  \style{#1}
  \centered
  \tightbaselines
  \tl
  \uppercase\expandafter{#2}
  \par
}}

\def\previewtitleblock{
  \vfil
  \previewtitleline{booktitle}{\the\previewtitle}
  \skipline
  \centerline{by}
  \skipline
  \previewtitleline{authorname}{\the\previewauthor}
  \vfil
}

\def\beginpreview{
  \nextrectopage
  \line{}
  \vfil
  \centerline{\sc Driscoll Brook Press}
  \centerline{proudly presents}
  \previewtitleblock
  \centerline{Turn the page for a preview}
  \vfil
  \output={\balancedspreads}
  \nextrectopage
  \sink{\chaptertitle{\the\previewtitle}}
}

\def\endpreview{
  \output={\facingpages}
  \nextrectopage
  \line{}
  \vfil
  \previewtitleblock
  \centerline{Available now at your favorite retailers}
  \vfil
  \centerline{\linelogo}
  \vfil
}
