\def\titlepageline#1{
  \raggedspine
  \tightbaselines
  \uppercase\expandafter{#1}
  \par
}

\def\halftitlepage{{
  \nextrectopage
  \booktitlefont
  \sink{\titlepageline{\the\title}}
  \vfil
}}

\def\titlepage{{
  \halftitlepage
  \authornamestyle
  \titlepageline{\the\author}
  \vfil
  \rightline{\rightlogo}
}}
