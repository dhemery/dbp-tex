\def\titlepageline#1#2{{
  \style{#1}
  \raggedspine
  \tightbaselines
  \tl
  \uppercase\expandafter{#2}
  \par
}}

\def\halftitlepage{
  \nextrectopage
  \sink{\titlepageline{booktitle}{\the\title}}
  \vfil
}

\def\titlepage{
  \halftitlepage
  \titlepageline{authorname}{\the\author}
  \vfil
  \rightline{\rightlogo}
}
