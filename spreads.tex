\def\beginbalancing{
  \output={\balancedspreads}
  \message{Balanced spreads begun on page \the\pageno }
  \gdef\afteroutput{}
}
\def\endbalancing{
  \output={\facingpages}
  \message{Facing pages begun on page \the\pageno }
  \gdef\afteroutput{}
}

\def\afteroutput{}

\def\balancedspreads{
  \facingpages
}

\def\beginspreads{
  \message{Balanced spreads requested on page \the\pageno}
  \nextpage
  \ifodd\pageno % Spreads must begin on an even page
    \gdef\afteroutput{\beginbalancing}
  \else
    \beginbalancing
  \fi
}

\def\endspreads{
  \message{Facing pages requested on page \the\pageno}
  \nextpage
  \ifodd\pageno % spreads must end on an odd page
    \gdef\afteroutput{\endbalancing}
  \else
    \endbalancing
  \fi
}
