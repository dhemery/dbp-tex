% utf8-t1 requires \bgroup and \loop
\let\bgroup={ \let\egroup=}
\def\loop#1\repeat{\def\body{#1}\iterate}
\def\iterate{\body \let\next\iterate \else\let\next\relax\fi \next}
\let\repeat=\fi

% Load the mapping from UTF-8 input to T1 output.
\input utf8-t1

% utf8-t1 maps certain UTF-8 characters to these control sequences.
\def\nbsp{\penalty10000\ }
\def\copyright{{\sym\char"A9}}
\def\registered{{\sym\char"AB}}
\def\trademark{{\sym\char"97}}
\def\ldots{\nbhairspace.\nbthinspace.\nbthinspace.}

% Macros for other basic characters
\def\'#1{{\accent"13 #1}}
\chardef\$=`\$
